Sub ExportToFile()
    Dim i As Integer
    Dim lastRow As Integer
    Dim myFile As String
    Dim myFilePath As String
    Dim myWorkbook As Workbook
    Dim myWorksheet As Worksheet
    
    Set myWorkbook = ActiveWorkbook
    Set myWorksheet = myWorkbook.ActiveSheet
    
    lastRow = myWorksheet.Cells(myWorksheet.Rows.Count, "A").End(xlUp).Row
    
    myFilePath = Application.ActiveWorkbook.Path & "\Output.txt"
    Open myFilePath For Output As #1
    
    For i = 1 To lastRow
        Write #1, "Item no: " & i
        Write #1, "Purpose: Party"
        Write #1, "Color: White"
        Write #1, "Item: " & myWorksheet.Range("A" & i).Value
    Next i
    
    Close #1
    
    MsgBox "Data exported to " & myFilePath, vbInformation
End Sub
